mkdir -p ~/.config/i3status
cp /etc/i3status.conf ~/.config/i3status/config

{
general {
        colors = true
        interval = 5
}

order += "ethernet _first_"
order += "battery all"
order += "tztime local"

ethernet _first_ {
        format_up = "E: %ip (%speed)"
        format_down = "E: down"
}

battery all {
        format = "%status %percentage %remaining"
}

tztime local {
        format = "%Y-%m-%d %H:%M:%S"
}
}
