3. sudo systemctl restart systemd-logind

{
#!/bin/bash

OLD_LAYOUT=$(setxkbmap -query | grep layout | awk '{print $2}')

setxkbmap us

FORK_FLAG=""
if [ "$1" = "nofork" ]; then
    FORK_FLAG="-n"
fi

i3lock $FORK_FLAG -c 111111

setxkbmap us,ru -option "grp:alt_shift_toggle"
}
